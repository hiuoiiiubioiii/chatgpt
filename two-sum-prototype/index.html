A compact static prototype that packages the core concepts studied for Q1 into a reusable analysis engine.

Problem identity

Input: nums, target

Output: indices [i, j] such that nums[i] + nums[j] = target and i != j.

value relation → position output
pair search → complement lookup

Input-output mapping

Raw input

nums   = [x0, x1, x2, ..., xn-1]
target = T

Raw output

[i, j]
nums[i] + nums[j] = T
current_value = nums[i]
needed_value  = target - current_value
if needed_value exists in seen map:
    return [seen[needed_value], i]
else:
    seen[current_value] = i

Invariant

Before processing index i, the map contains all values from indices [0 .. i-1] mapped to valid earlier indices.

processed prefix | current | unprocessed suffix

This invariant makes the first hit immediately correct.

Google-style C++ reference

#include <unordered_map>
#include <vector>

class Solution {
 public:
  std::vector<int> twoSum(const std::vector<int>& nums, int target) {
    std::unordered_map<int, int> seen_index_by_value;

    for (int i = 0; i < static_cast<int>(nums.size()); ++i) {
      const int current_value = nums[i];
      const int needed_value = target - current_value;

      const auto it = seen_index_by_value.find(needed_value);
      if (it != seen_index_by_value.end()) {
        return {it->second, i};
      }

      seen_index_by_value[current_value] = i;
    }

    return {};
  }
};
